// --------------------------------------------------------------------------------
// 3. IMPORTANT PLAYWRIGHT COMMANDS
// --------------------------------------------------------------------------------

/*
Initialize Playwright Project:
    npm init playwright@latest

Run All Tests:
    npx playwright test

Run Specific Test File:
    npx playwright test techx.spec.js

Run Tests in UI Mode:
    npx playwright test --ui

Run Tests in Headed Mode:
    npx playwright test --headed

Run Tests in Debug Mode:
    npx playwright test --debug

Generate Automation Code:
    npx playwright codegen

View HTML Report:
    npx playwright show-report
*/

*/


// --------------------------------------------------------------------------------
// 6. CREATING TEST CASES
// --------------------------------------------------------------------------------

/*
Example:

test('check title', async ({ page }) => {

});

Explanation:
- 'check title' → Name of test case
- async → Allows asynchronous execution
- page → Browser page object used for browser interactions
*/


// --------------------------------------------------------------------------------
// 7. OPENING A WEBSITE
// --------------------------------------------------------------------------------

/*
Example:

await page.goto("https://www.techxincorporation.com/");

Explanation:
- goto() opens the specified URL
- await ensures execution waits until page loads completely
*/


// --------------------------------------------------------------------------------
// 8. LOCATORS IN PLAYWRIGHT
// --------------------------------------------------------------------------------

/*
Example:

const consultBtn =
page.getByRole("link", { name: /Request/ }).nth(1);

Explanation:
- getByRole() locates elements using accessibility roles
- "link" refers to anchor/link element
- { name: /Request/ } uses regex matching
- nth(1) selects the second matching element
*/


// --------------------------------------------------------------------------------
// 9. CLICK ACTIONS
// --------------------------------------------------------------------------------

/*
Example:

await consultBtn.click();

Explanation:
- click() performs click operation on located element
- await ensures action completes before moving forward
*/


// --------------------------------------------------------------------------------
// 10. ASSERTIONS / VALIDATIONS
// --------------------------------------------------------------------------------

/*
Example:

expect(page.url()).toBe(
"https://www.techxincorporation.com/contact"
);

Explanation:
- expect() validates application behavior
- toBe() compares actual and expected values
- Used for verification in automation testing
*/
